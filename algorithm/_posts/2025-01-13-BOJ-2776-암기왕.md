public static int binarySearch(int[] nums, int target, int start, int end) {
    if (start > end) {
        return 0;
    }

    int mid = (start + end) / 2;
    if (target == nums[mid]) {
        return 1;
    } else if (target > nums[mid]) {
        return binarySearch(nums, target, mid + 1, end);
    } else if (target < nums[mid]) {
        return binarySearch(nums, target, start, mid - 1);
    }
    return 0;
}
